 21class PhysicsManager:
 22    """
 23    Manages the physics simulation of objects.
 24
 25    Handles adding objects to the physics space, updating physics simulation steps,
 26    and configuring object shapes such as rectangles and circles.
 27    """
 28
 29    def __init__(self) -> None:
 30        """
 31        Initializes a PhysicsManager
 32        """
 33        self.space: pymunk.Space = pymunk.Space()
 34        """The main space of the objects"""
 35
 36        self.space.on_collision(
 37            None,None,
 38            begin=self._on_collision_begin,
 39            pre_solve=self._on_collision_pre_solve,
 40            separate=self._on_collision_separate,
 41        )
 42
 43    def add(self, Object: "Object") -> None:
 44        """
 45        Add an Object to the physics simulation.
 46
 47        Args:
 48            Object (Object): The object to add to the physics space.
 49        """
 50        space = self.space
 51        physics = Object.physics
 52        static = Object.static
 53        shape_type = Object.shape_type
 54
 55        if shape_type == 1:
 56            self.add_rect(Object, space, static, **physics)
 57        elif shape_type == 2:
 58            self.add_circle(Object, space, static, **physics)
 59        else:
 60            engine.error(RuntimeError("Unknown shape type"))
 61
 62    def update(self, step: int) -> None:
 63        """
 64        Advance the physics simulation by a specified number of steps.
 65
 66        Args:
 67            step (int): The amount of time steps to advance the simulation.
 68        """
 69        self.space.step(step)

 71    @staticmethod
 72    def add_rect(Object: "Object", space: pymunk.Space, static: bool, **rigidshape_kwargs) -> None:
 73        """
 74        Add a rectangular object to the physics space.
 75
 76        Args:
 77            Object (Object): The object to add.
 78            static (bool): Whether the object is static (non-movable).
 79            space (pymunk.Space): The physics space to add the object to.
 80            **rigidshape_kwargs: Additional rigid shape properties to set on the pymunk shape.
 81        """
 82        body_type = pymunk.Body.STATIC if static else pymunk.Body.DYNAMIC
 83        rigidbody = pymunk.Body(body_type=body_type)
 84        rigidbody.position = (
 85            Object.spawn[0] + Object.size[0] / 2,
 86            Object.spawn[1] + Object.size[1] / 2
 87        )
 88
 89        rigidshape = pymunk.Poly.create_box(rigidbody, Object.size)
 90        for key, value in rigidshape_kwargs.items():
 91            setattr(rigidshape, key, value)
 92
 93        space.add(rigidbody, rigidshape)
 94        Object.rigidbody = rigidbody
 95        Object.rigidshape = rigidshape
 96
 97        # link shape back to Object for collision callbacks
 98        rigidshape.pyxora_object = Object
 99
100    @staticmethod
101    def add_circle(Object: "Object", space: pymunk.Space, static: bool, **rigidshape_kwargs) -> None:
102        """
103        Add a circular object to the physics space.
104
105        Args:
106            Object (Object): The object to add.
107            static (bool): Whether the object is static (non-movable).
108            space (pymunk.Space): The physics space to add the object to.
109            **rigidshape_kwargs: Additional rigid shape properties to set on the pymunk shape.
110        """
111        body_type = pymunk.Body.STATIC if static else pymunk.Body.DYNAMIC
112        rigidbody = pymunk.Body(body_type=body_type)
113        rigidbody.position = Object.spawn
114
115        rigidshape = pymunk.Circle(body=rigidbody, radius=Object.size)
116        for key, value in rigidshape_kwargs.items():
117            setattr(rigidshape, key, value)
118
119        space.add(rigidbody, rigidshape)
120        Object.rigidbody = rigidbody
121        Object.rigidshape = rigidshape
122
123        # link shape back to Object for collision callbacks
124        rigidshape.pyxora_object = Object

126    def _collision_objects_from_arbiter(self,arbiter: pymunk.Arbiter) -> tuple[Optional["Object"], Optional["Object"]]:
127        """Helper to get our Object instances from a pymunk arbiter."""
128        shapes = arbiter.shapes
129        if len(shapes) != 2:
130            return None, None
131
132        shape_a, shape_b = shapes
133        obj_a = getattr(shape_a, "pyxora_object", None)
134        obj_b = getattr(shape_b, "pyxora_object", None)
135        return obj_a, obj_b
136
137    def _on_collision_begin(self,arbiter: pymunk.Arbiter,space: pymunk.Space,data: Any) -> bool:
138        """
139        Called once when two shapes start touching this step.
140        """
141        obj_a, obj_b = self._collision_objects_from_arbiter(arbiter)
142        if obj_a is None or obj_b is None:
143            return False
144
145        obj_a._on_collision_enter_scripts(obj_b)
146        obj_b._on_collision_enter_scripts(obj_a)
147        return True
148    def _on_collision_pre_solve(self,arbiter: pymunk.Arbiter,space: pymunk.Space,data: Any) -> bool:
149        """
150        Called every step while two shapes are touching.
151        """
152        obj_a, obj_b = self._collision_objects_from_arbiter(arbiter)
153        if obj_a is None or obj_b is None:
154            return False
155
156        obj_a._on_collision_scripts(obj_b)
157        obj_b._on_collision_scripts(obj_a)
158        return True
159    def _on_collision_separate(self,arbiter: pymunk.Arbiter,space: pymunk.Space,data: Any) -> bool:
160        """
161        Called once when two shapes stop touching.
162        """
163        obj_a, obj_b = self._collision_objects_from_arbiter(arbiter)
164        if obj_a is None or obj_b is None:
165            return False
166
167        obj_a._on_collision_exit_scripts(obj_b)
168        obj_b._on_collision_exit_scripts(obj_a)
169        return True
170class Objects:
171    """
172    Manages a collection of Object instances in a Scene.
173
174    Responsible for adding/removing objects, updating physics and scripts,
175    drawing objects, and toggling debug hitboxes.
176    """
177
178    def __init__(self,scene:"Scene") -> None:
179        """
180        Initialize an Objects manager.
181
182        Args:
183            scene ("Scene"): Renderer to use.
184        """
185        self.Physics:PhysicsManager  = PhysicsManager()
186
187        self.__data: List["Object"] = []
188        self.__counter: int = 0
189
190        self.__scene = scene
191        self.__render = scene.camera
192
193    def __iter__(self) -> Iterator["Object"]:
194        """Iterate over all managed objects."""
195        return iter(self.__data)
196
197    @property
198    def total(self) -> int:
199        """Get the total objects."""
200        return self.__counter
201
202    def add(self, Object: "Object") -> None:
203        """
204        Add an Object to the manager and physics simulation.
205
206        Args:
207            Object (Object): The object to add.
208        """
209        self.__counter += 1
210
211        Object.id = self.__counter
212        Object.Manager = self
213
214        self.Physics.add(Object)
215        self.__data.append(Object)
216
217        for script in Object.scripts:
218            script._start(Object)
219
220    def remove(self, obj: "Object") -> None:
221        """
222        Remove an object from the manager without firing kill callbacks.
223
224        Args:
225            obj (Object): The object to remove.
226        """
227        # remove from physics space
228        space = self.Physics.space
229        space.remove(obj.rigidshape)
230        space.remove(obj.rigidbody)
231
232        obj in self.__data and self.__data.remove(obj)
233
234    def pop(self, index: int) -> "Object":
235        """
236        Remove and return an object at a specified index.
237
238        Args:
239            index (int): Index of the object to remove.
240        """
241        self.__data[index].kill()
242
243    def clear(self) -> None:
244        """Remove all objects and reset the physics manager."""
245        self.Physics = PhysicsManager()
246
247        self.__counter = 0
248        self.__data.clear()
249
250    def set_gravity(self, gravity: Tuple[int | float, int | float] | pygame.math.Vector2 | pygame.math.Vector3) -> None:
251        """
252        Set the gravity for the physics simulation.
253
254        Args:
255            gravity (Tuple[int | float, int | float] | pygame.math.Vector2 | pygame.math.Vector3)): Gravity vector (x, y).
256        """
257        self.Physics.space.gravity = gravity
258
259    def update(self) -> None:
260        """Update physics and object scripts each frame."""
261        dt = self.__scene.dt
262        self.Physics.update(dt)
263        for obj in self:
264            obj.update(dt)
265
266    def draw(self) -> None:
267        """Draw all visible objects and their hitboxes (if enabled)."""
268        for obj in self:
269            obj.draw(self.__render)

272class Object:
273    """
274    Represents a single game object with optional physics and scripts.
275
276    Supports rectangle or circle shapes, images, physics properties, and custom scripts.
277    """
278
279    def __init__(self, pos, size, image: Optional[Image] = None, shape_type: int = 1) -> None:
280        """
281        Initialize a game object.
282
283        Args:
284            pos (tuple): The object position (x, y). The code will internally adjust it so
285                objects are positioned in the center of the cords.
286            size (tuple or int): Size of the object.
287            image (Image, optional): Image to render. Defaults to engine icon.
288            shape_type (int, optional): The object shape type
289        """
290        pos = (pos[0] - size, pos[1] - size) if shape_type == 2 else (pos[0] - size[0] / 2, pos[1] - size[1] / 2)
291        img_size = (size * 2, size * 2) if shape_type == 2 else size
292        if image:
293            img = Image(
294                image,
295                pos,
296                custom_size=img_size,
297                shape_type=shape_type
298            )
299        else:
300            img = Image(
301                Assets.get("engine", "images", "icon"),
302                pos,
303                custom_size=img_size,
304                shape_type=shape_type
305            )
306
307        self.Manager: Optional[Objects] = None
308        self.id: Optional[int] = None
309
310        self.__rigidbody: Optional[pymunk.Body] = None
311        self.__rigidshape: Optional[pymunk.Shape] = None
312
313        self.__image: Image = img
314        self.__spawn = pos
315        self.__size = size
316        self.__shape_type = shape_type
317        self.__invisible = False
318        self.__scripts: List["Script"] = []
319
320        # Every object need to have phyiscs properties
321        # add to init the default static values
322        self.add_physics(static=True)
323
324    @property
325    def rigidbody(self) -> Optional[pymunk.Body]:
326        """Get the object physics rigidbody."""
327        return self.__rigidbody
328
329    @rigidbody.setter
330    def rigidbody(self, value: pymunk.Body) -> None:
331        """Set the object physics rigidbody."""
332        self.__rigidbody = value
333
334    @property
335    def rigidshape(self) -> Optional[pymunk.Shape]:
336        """Get the object physics collision shape."""
337        return self.__rigidshape
338
339    @rigidshape.setter
340    def rigidshape(self, value: pymunk.Shape) -> None:
341        """Set the object physics collision shape."""
342        self.__rigidshape = value
343
344    @property
345    def image(self) -> Image:
346        """Get the object render image."""
347        return self.__image
348
349    @property
350    def spawn(self) -> Tuple[float, float]:
351        """Get the object starting position."""
352        return self.__spawn
353
354    @property
355    def size(self) -> Union[Tuple[int, int], int]:
356        """Get the object size."""
357        return self.__size
358
359    @property
360    def shape(self) -> str:
361        """Get the object shape."""
362        return object_shapes[self.__shape_type]
363
364    @property
365    def shape_type(self) -> int:
366        """Get the object shape type."""
367        return self.__shape_type
368
369    @property
370    def invisible(self) -> bool:
371        """Get whether the object is invisible."""
372        return self.__invisible
373
374    @invisible.setter
375    def invisible(self, value: bool) -> None:
376        """Set the object invisible state."""
377        self.__invisible = value
378
379    @property
380    def scripts(self) -> List["Script"]:
381        """Get the list of scripts attached to the object."""
382        return self.__scripts
383
384    @property
385    def hitbox(self) -> Union[Rect, Circle]:
386        """Return a visual representation of the object's hitbox."""
387        if self.shape_type == 1:
388            return Rect(self.position, self.size, "red")
389        elif self.shape_type == 2:
390            return Circle(self.position, self.size, "red")
391        else:
392            engine.error(RuntimeError("Unknown shape type"))
393
394    @property
395    def position(self) -> pygame.math.Vector2:
396        """Return the object position."""
397        pos = vector(self.rigidbody.position[0], self.rigidbody.position[1])
398        if self.shape_type == 1:
399            pos.x -= self.size[0] / 2
400            pos.y -= self.size[1] / 2
401        return pos
402
403    def move(self, new_pos: Tuple[float, float]) -> None:
404        """
405        Move the object relative to its current position.
406
407        Args:
408            new_pos (tuple): Delta position (dx, dy) to move.
409        """
410        pos = self.rigidbody.position
411        self.move_at((pos[0] + new_pos[0], pos[1] + new_pos[1]))
412
413    def move_at(self, new_pos: Tuple[float, float]) -> None:
414        """
415        Move the object to an absolute position.
416
417        Args:
418            new_pos (tuple): Absolute position (x, y).
419        """
420        self.rigidbody.position = new_pos
421        self.Manager.Physics.space.reindex_shapes_for_body(self.rigidbody)
422        self.static and self.Manager.Physics.space.reindex_static()
423
424    def add_physics(self, mass: float = 1, friction: float = 0, elasticity: float = 0, static: bool = False) -> None:
425        """
426        Add the physics properties for the object
427        Note: This properties can't change after the Object is added in Objects.
428
429        Args:
430            mass (float, optional): Mass of the object. Defaults to 1.
431            friction (float, optional): Friction coefficient. Defaults to 0.
432            elasticity (float, optional): Elasticity coefficient. Defaults to 0.
433            static (bool, optional): Whether the object is static. Defaults to False.
434        """
435        shape_type = self.__shape_type
436        self.physics = {
437            "mass": mass,
438            "friction": friction,
439            "elasticity": elasticity,
440            "collision_type": shape_type,
441        }
442        self.static = static
443
444    def add_script(self, script_cls: type["Script"]) -> None:
445        """
446        @public
447        Attach a script to this object.
448
449        Scripts are executed in the order they are added.
450        """
451        script = script_cls()
452        self.__scripts.append(script)
453
454    def add_scripts(self, *script_classes: type["Script"]) -> None:
455        """
456        Attach multiple scripts to this object.
457
458        Usage:
459            obj.add_scripts(ScriptA, ScriptB, ScriptC)
460        """
461        for script_cls in script_classes:
462            self.add_script(script_cls)
463
464    def kill(self) -> None:
465        """
466        @public
467        Destroy this object.
468
469        Removes the object from its manager and notifies all attached scripts
470        that it is being killed.
471        """
472        if self.Manager is None:
473            return
474        # notify scripts
475        self._on_kill_scripts()
476        self.Manager.remove(self)
477
478    def update(self, dt: float) -> None:
479        """
480        @private
481        Update the object image position and scripts.
482        """
483        if self.image:
484            self.image.move_at(self.rigidbody.position)
485            if self.shape_type == 1:
486                self.image.move((-self.size[0] / 2, -self.size[1] / 2))
487            elif self.shape_type == 2:
488                self.image.move((-self.size, -self.size))
489
490        self._update_scripts(dt)
491
492    def draw(self, render) -> None:
493        """
494        @private
495        Draw the Object
496        """
497        if self.invisible:
498            return
499        render.draw_image(self.__image)
500        pyxora.debug and render.draw_shape(self.hitbox, 1)
501
502        self._draw_scripts()
503
504    def _update_scripts(self, dt: float) -> None:
505        """
506        @private
507        Update all scripts attached to this object.
508        """
509        for s in self.__scripts:
510            s._update(self, dt)
511
512    def _draw_scripts(self) -> None:
513        """
514        @private
515        Draw hooks for all scripts attached to this object.
516        """
517        for s in self.__scripts:
518            s._draw(self)
519
520    def _on_kill_scripts(self) -> None:
521        """
522        @private
523        Notify all scripts that this object is being killed.
524        """
525        for s in self.__scripts:
526            s._on_kill(self)
527
528    def _on_collision_scripts(self, other: "Object") -> None:
529        """
530        @private
531        Notify all scripts that a collision with another object is ongoing.
532
533        Args:
534            other (Object): The other object currently colliding with this one.
535        """
536        for s in self.__scripts:
537            s._on_collision(self, other)
538
539    def _on_collision_enter_scripts(self, other: "Object") -> None:
540        """
541        @private
542        Notify all scripts that a collision with another object just began.
543
544        Args:
545            other (Object): The other object that has just started colliding
546                with this one.
547        """
548        for s in self.__scripts:
549            s._on_collision_enter(self, other)
550
551    def _on_collision_exit_scripts(self, other: "Object") -> None:
552        """
553        @private
554        Notify all scripts that a collision with another object just ended.
555
556        Args:
557            other (Object): The other object that has just stopped colliding
558                with this one.
559        """
560        for s in self.__scripts:
561            s._on_collision_exit(self, other)
