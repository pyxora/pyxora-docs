 56        scenes: str = None,scripts: str = None,

 58        **custom_types: str
 59    ) -> None:
 60        """
 61        Initialize the Assets system by loading asset files into the Data structure.
 62
 63        Args:
 64            pre_load (bool): Whether to preload the assets immediately. Defaults to True.
 65            images (str, optional): Path to image files.
 66            fonts (str, optional): Path to font files.

 75        cls.engine.fonts.update(cls.__get_default_font())  # add default font to engine

 77        cls._load_data_files(
 78            caller,
 79            images,fonts,
 80            scenes,scripts,
 81            music,sfx,
 82            **custom_types
 83        )

 86    @classmethod
 87    def get(cls, *loc: str) -> Any:
 88        """
 89        Retrieve a nested asset value by Assets data structure.
 90
 91        Args:
 92            *loc (str): Path to the desired asset.
 93                - The first item is the source type ('data', 'engine').
 94                - If omitted then 'data' is used by default.
 95                - All items are normalized to lowercase.
 96
 97        Returns:
 98            Any: The value found at the specified path.
 99
100        Raises:
101            KeyError: If no path is provided (loc is empty) or if any lookup fails.
102
103        Example:
104            Assets.get("data", "images", "player") # Retrieve the player image \n
105            Assets.get("images", "player") # Shortcut: source defaults to 'data' \n
106            Assets.get("custom", "text", "my_text") # Retrieve a custom text \n
107            Assets.get("engine", "images", "icon") # Retrieve an engine asset \n
108        """
109
110        if not loc:
111            raise KeyError("Asset lookup failed, no asset path provided")
112
113        # Default to 'data' if not explicitly set
114        source = "data"
115        keys = loc
116
117        if loc[0].lower() in ("data", "engine"):
118            source = loc[0].lower()
119            keys = loc[1:]
120
121        data = getattr(cls, source)
122        for key in keys:
123            key = key.lower()
124            # If data is a dict, use key access (lower)
125            if isinstance(data, dict):
126                data = data.get(key)
127            else:
128                # Otherwise, try attribute access
129                data = getattr(data, key, None)
130            if data is None:
131                raise KeyError(f"Could not find asset at path: {'->'.join(loc)}")
132
133        return data

136    def load(cls, source: "str") -> None:
137        """
138        Load file paths into the data system
139
140        Args:
141            source (str): The data name to load data from.
142        """
143
144        data = getattr(cls, source)

146            file_dict = data.files.get(category)
147            if not file_dict:
148                continue
149
150            # Built-in categories
151            if hasattr(data, category):
152                asset_store = getattr(data, category)
153            else:
154                if category not in data.custom:
155                    data.custom[category] = {}
156                asset_store = data.custom[category]
157
158            # Store the loaded values in Data
159            for name, path in file_dict.items():
160                asset_store[name] = loader(path)

250    @staticmethod
251    def __get_all_files(path,ignore=None) -> dict[str, dict[str, str]]:
252        """
253        Recursively scan provided directories and build a nested dictionary of file paths.
254
255        Args:
256            path (dict[str, str]): A dictionary where each key is an asset type
257                                (like "images" or "fonts") and the value is the path to its folder.
258            ignore (set[str], optional): A set of folder names to exclude from scanning.
259                                        "__pycache__" is always ignored.
260
261        Returns:
262            dict[str, dict[str, str]]: A nested dictionary structured like:
263                {
264                    "images": {
265                        "player": "/path/to/images/player.png",
266                        "enemy": "/path/to/images/enemy.png"
267                    },
268                    "fonts": {
269                        "main": "/path/to/fonts/arial.ttf"
270                    },
271                    ...
272                }
273        """
274        if not ignore:
275            ignore = set()
276
277        ignore.update({"__pycache__"})
278
279        data = {}
280        for key,value in path.items():
281            for root, dirs, files in os.walk(value,topdown=False):
282                ftype = os.path.basename(root)
283                if ftype in ignore:
284                    continue
285                data[key] = {}
286                for file in files:
287                    full_path = os.path.join(root, file)
288                    name,_ = os.path.splitext(os.path.basename(full_path))
289                    data[key][name] = full_path
290        return data
