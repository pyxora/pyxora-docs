151    def create(self,name: str,state: int = 0,**kwargs: Any) -> int:
152        """
153        Create and store a new custom event.
154
155        Args:
156            name (str): The name of the event.
157            state (int): The event state, where:
158                - 1: Runtime
159                - -1: Pause time
160                - 0: Both (default = 0)
161            **kwargs: Additional arguments passed to the event's info. Can be any extra data needed for the event.
162
163        Returns:
164            int: The ID of the new event.
165        """
166        event_id = pygame.event.custom_type()
167        create_time = self._now(state)
168        last_time = create_time
169        basic_argv = {
170            "name":name,"custom":True,
171            "create_time":create_time,"calls":0,
172            "last_time":last_time,"timer":None,
173            "loops":None,"state":state
174        }
175
176        kwargs.update(basic_argv)
177        event = pygame.event.Event(
178            event_id,
179            kwargs
180        )
181
182        self.__data[name] = event
183        return event_id

250    # Update all the scene events and it properties
251    def update(self,state: int):
252        """
253        @private
254
255        (Not include in the doc as it should be call only inside scene itself)
256        Updates all the custom events and it's properties at the scene state.
257
258        Args:
259            state (int): The scene state. (default = -1, forever).
260        """
261        for event_name in self.__data:
262            event = self.get(event_name)
263
264            if not (self._is_state(state,event.state)):
265                continue
266
267            if not event.timer:
268                continue
269
270            if event.calls == event.loops:
271                continue
272
273            self._update_time(event)

275    def _now(self, state: int) -> float:
276        """Returns the current time based on the state.
277
278        Args:
279            state (int): Determines which time value to return.
280                - If 0: Returns runtime + pausetime.
281                - If >0: Returns runtime.
282                - If <0: Returns pausetime.
283
284        Returns:
285            float: The calculated current time value.
286        """
287        runtime = self.__scene.runtime
288        pausetime = self.__scene.pausetime
289        if state == 0:
290            return runtime + pausetime
291
292        now = runtime if state > 0 else pausetime
293        return now

295    def _update_time(self, event:pygame.event.Event) -> None:
296        """Checks if an event should be triggered based on elapsed time.
297
298        If the time since the last event trigger exceeds the timer threshold,
299        the event is posted and its last_time is updated.
300
301        Args:
302            event (object): An object with the following attributes:
303                - state (int): State used to determine which time value to use.
304                - last_time (float): Timestamp of the last event trigger.
305                - timer (float): Timer threshold in milliseconds.
306                - name (str): Name of the event to post.
307
308        Side Effects:
309            Posts the event via self.post() if the condition is met.
310            Updates event.last_time.
311        """
312        now = self._now(event.state)
313        diff = (now - event.last_time) * 1000
314        is_time = diff >= event.timer
315        if not is_time:
316            return
317
318        self.post(event.name)
319        event.last_time = now
320        event.calls += 1

322    @staticmethod
323    def _is_state(state: int, event_state: int) -> bool:
324        """Checks if the event state matches the given state or is neutral (0).
325
326        Args:
327            state (int): Target state to compare against.
328            event_state (int): Current state of the event.
329
330        Returns:
331            bool: True if the event_state equals the state or is 0, False otherwise.
332        """
333        same = event_state == state
334        is_zero = event_state == 0
335        return (same or is_zero)
